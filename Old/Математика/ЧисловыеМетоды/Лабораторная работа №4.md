## 1. –¢–µ–æ—Ä–µ—Ç–∏—á–µ—Å–∫–∞—è —á–∞—Å—Ç—å

–ú–µ—Ç–æ–¥ –ì–∞—É—Å—Å–∞ –∑–∞–∫–ª—é—á–∞–µ—Ç—Å—è –≤ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ–º –∏—Å–∫–ª—é—á–µ–Ω–∏–∏ –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω—ã—Ö, –ø—Ä–∏–≤–æ–¥—è —Å–∏—Å—Ç–µ–º—É –∫ —Ç—Ä–µ—É–≥–æ–ª—å–Ω–æ–º—É –≤–∏–¥—É, –∞ –∑–∞—Ç–µ–º –≤—ã–ø–æ–ª–Ω—è—è –æ–±—Ä–∞—Ç–Ω—ã–π —Ö–æ–¥.

## 6. –ó–∞–∫–ª—é—á–µ–Ω–∏–µ

–ú–µ—Ç–æ–¥ –ì–∞—É—Å—Å–∞ –ø–æ–¥—Ç–≤–µ—Ä–¥–∏–ª –≤—ã—Å–æ–∫—É—é —É—Å—Ç–æ–π—á–∏–≤–æ—Å—Ç—å –∏ —Ç–æ—á–Ω–æ—Å—Ç—å –ø—Ä–∏ —Ä–µ—à–µ–Ω–∏–∏ —Å–∏—Å—Ç–µ–º –ª–∏–Ω–µ–π–Ω—ã—Ö —É—Ä–∞–≤–Ω–µ–Ω–∏–π.  
–ú–µ—Ç–æ–¥ –Ø–∫–æ–±–∏ –≤ –¥–∞–Ω–Ω–æ–º –≤–∞—Ä–∏–∞–Ω—Ç–µ –Ω–µ —É–¥–æ–≤–ª–µ—Ç–≤–æ—Ä—è–µ—Ç —É—Å–ª–æ–≤–∏—é —Å—Ö–æ–¥–∏–º–æ—Å—Ç–∏, —á—Ç–æ –∏–ª–ª—é—Å—Ç—Ä–∏—Ä—É–µ—Ç –≤–∞–∂–Ω–æ—Å—Ç—å –ø—Ä–µ–¥–≤–∞—Ä–∏—Ç–µ–ª—å–Ω–æ–≥–æ –∞–Ω–∞–ª–∏–∑–∞ –º–∞—Ç—Ä–∏—Ü—ã —Å–∏—Å—Ç–µ–º—ã.  
–ü–æ–ª—É—á–µ–Ω–Ω—ã–µ –æ—Ü–µ–Ω–∫–∏ –ø–æ–≥—Ä–µ—à–Ω–æ—Å—Ç–µ–π –ø–æ–∫–∞–∑—ã–≤–∞—é—Ç, —á—Ç–æ –ø—Ä—è–º–æ–µ —Ä–µ—à–µ–Ω–∏–µ –æ–±–ª–∞–¥–∞–µ—Ç —Ö–æ—Ä–æ—à–µ–π —á–∏—Å–ª–µ–Ω–Ω–æ–π —É—Å—Ç–æ–π—á–∏–≤–æ—Å—Ç—å—é.


–†–µ–∞–ª–∏–∑–∞—Ü–∏—è –Ω–∞ Python

```python
import numpy as np

  

# ======================================================

# –õ–∞–±–æ—Ä–∞—Ç–æ—Ä–Ω–∞—è —Ä–∞–±–æ—Ç–∞ ‚Ññ4

# –†–µ—à–µ–Ω–∏–µ —Å–∏—Å—Ç–µ–º—ã –ª–∏–Ω–µ–π–Ω—ã—Ö —É—Ä–∞–≤–Ω–µ–Ω–∏–π

# –ú–µ—Ç–æ–¥—ã: –ì–∞—É—Å—Å–∞ –∏ –Ø–∫–æ–±–∏ (–≤–∞—Ä–∏–∞–Ω—Ç 1)

# ======================================================

  

# –ò—Å—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ (–≤–∞—Ä–∏–∞–Ω—Ç 1)

A = np.array([

[0.946, 0.628, 3.215, 0.194],

[0.213, 0.429, 0.997, 2.351],

[0.021, 0.375, 0.326, 0.273],

[0.425, 0.281, 0.519, 0.207]

], dtype=float)

  

b = np.array([0.412, 0.297, 3.416, 0.003], dtype=float)

  

print("=" * 70)

print("–õ–∞–±–æ—Ä–∞—Ç–æ—Ä–Ω–∞—è —Ä–∞–±–æ—Ç–∞ ‚Ññ4")

print("–†–µ—à–µ–Ω–∏–µ —Å–∏—Å—Ç–µ–º—ã –ª–∏–Ω–µ–π–Ω—ã—Ö —É—Ä–∞–≤–Ω–µ–Ω–∏–π (–≤–∞—Ä–∏–∞–Ω—Ç 1)")

print("=" * 70)

  

# ------------------------------------------------------

# 1. –ò—Å—Ö–æ–¥–Ω–∞—è —Å–∏—Å—Ç–µ–º–∞

# ------------------------------------------------------

print("\n1Ô∏è‚É£ –ò—Å—Ö–æ–¥–Ω–∞—è —Å–∏—Å—Ç–µ–º–∞ —É—Ä–∞–≤–Ω–µ–Ω–∏–π:")

for i in range(len(A)):

eq = " + ".join([f"{A[i, j]:.3f}x{j+1}" for j in range(len(A[i]))])

print(f" {eq} = {b[i]:.3f}")

  

# ------------------------------------------------------

# 2. –ú–µ—Ç–æ–¥ –ì–∞—É—Å—Å–∞

# ------------------------------------------------------

x_gauss = np.linalg.solve(A, b)

  

print("\n2Ô∏è‚É£ –†–µ—à–µ–Ω–∏–µ —Å–∏—Å—Ç–µ–º—ã –º–µ—Ç–æ–¥–æ–º –ì–∞—É—Å—Å–∞:")

for i, xi in enumerate(x_gauss, 1):

print(f" x{i} = {xi: .6f}")

  

# ------------------------------------------------------

# 3. –û–±—Ä–∞—Ç–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞ –∏ –Ω–æ—Ä–º—ã

# ------------------------------------------------------

A_inv = np.linalg.inv(A)

norm_A = np.max(np.sum(np.abs(A), axis=0))

norm_Ainv = np.max(np.sum(np.abs(A_inv), axis=0))

  

print("\n3Ô∏è‚É£ –û–±—Ä–∞—Ç–Ω–∞—è –º–∞—Ç—Ä–∏—Ü–∞ A‚Åª¬π:")

for row in A_inv:

print(" ", " ".join(f"{val: .6f}" for val in row))

  

print(f"\n ||A|| = {norm_A:.3f}")

print(f" ||A‚Åª¬π|| = {norm_Ainv:.3f}")

  

# ------------------------------------------------------

# 4. –ü–æ–≥—Ä–µ—à–Ω–æ—Å—Ç–∏

# ------------------------------------------------------

Œîb = 1e-3

b_norm = np.max(np.abs(b))

abs_error = norm_Ainv * Œîb

rel_error = norm_Ainv * norm_A * (Œîb / b_norm)

  

print("\n4Ô∏è‚É£ –û—Ü–µ–Ω–∫–∞ –ø–æ–≥—Ä–µ—à–Ω–æ—Å—Ç–µ–π:")

print(f" –ê–±—Å–æ–ª—é—Ç–Ω–∞—è –ø–æ–≥—Ä–µ—à–Ω–æ—Å—Ç—å —Å–≤–æ–±–æ–¥–Ω—ã—Ö —á–ª–µ–Ω–æ–≤: Œîb = {Œîb}")

print(f" –ê–±—Å–æ–ª—é—Ç–Ω–∞—è –ø–æ–≥—Ä–µ—à–Ω–æ—Å—Ç—å —Ä–µ—à–µ–Ω–∏—è: Œîx ‚â§ {abs_error:.6f}")

print(f" –û—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–∞—è –ø–æ–≥—Ä–µ—à–Ω–æ—Å—Ç—å: Œîx/x ‚â§ {rel_error:.6f} (~{rel_error*100:.3f}%)")

  

# ------------------------------------------------------

# 5. –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –∫ –º–µ—Ç–æ–¥—É –Ø–∫–æ–±–∏

# ------------------------------------------------------

D = np.diag(np.diag(A))

D_inv = np.linalg.inv(D)

B = np.eye(4) - D_inv @ A

c = D_inv @ b

norm_B = np.max(np.sum(np.abs(B), axis=0))

  

print("\n5Ô∏è‚É£ –ü–æ–¥–≥–æ—Ç–æ–≤–∫–∞ –∫ –∏—Ç–µ—Ä–∞—Ü–∏–æ–Ω–Ω–æ–º—É –º–µ—Ç–æ–¥—É –Ø–∫–æ–±–∏:")

print(" –í–µ–∫—Ç–æ—Ä c = D‚Åª¬π¬∑b:")

print(" ", np.round(c, 6))

print("\n –ú–∞—Ç—Ä–∏—Ü–∞ B = I - D‚Åª¬π¬∑A:")

for row in B:

print(" ", " ".join(f"{val: .3f}" for val in row))
````
