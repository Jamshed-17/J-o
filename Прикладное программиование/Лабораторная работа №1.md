## 1. Создание базы данных

```sql
CREATE DATABASE Employee;

```

## 2. Создание таблиц

```sql
-- Общие сведения (преподаватели)
CREATE TABLE Teachers (
    TeacherID INT PRIMARY KEY,
    FullName VARCHAR(100) NOT NULL,
    Department VARCHAR(100) NOT NULL,
    Address VARCHAR(150) NOT NULL
);

-- Педиагруппа (нагрузка)
CREATE TABLE Workload (
    TeacherID INT PRIMARY KEY,
    Tariff DECIMAL(10,2) NOT NULL,
    HoursPerYear INT NOT NULL,
    FOREIGN KEY (TeacherID) REFERENCES Teachers(TeacherID)
);

-- Дисциплины
CREATE TABLE Disciplines (
    SubjectID INT PRIMARY KEY,
    SubjectName VARCHAR(100) NOT NULL
);

-- Предметы (преподавание)
CREATE TABLE Teaching (
    TeacherID INT,
    GroupName VARCHAR(50),
    SubjectID INT,
    PRIMARY KEY (TeacherID, GroupName, SubjectID),
    FOREIGN KEY (TeacherID) REFERENCES Teachers(TeacherID),
    FOREIGN KEY (SubjectID) REFERENCES Disciplines(SubjectID)
);
```

## 4. Проверка целостности данных

```sql
-- Проверка внешних ключей
SELECT 
    t.TeacherID,
    t.FullName,
    w.Tariff,
    w.HoursPerYear
FROM Teachers t
INNER JOIN Workload w ON t.TeacherID = w.TeacherID;

-- Проверка связей преподавания
SELECT 
    t.FullName,
    te.GroupName,
    d.SubjectName
FROM Teaching te
JOIN Teachers t ON te.TeacherID = t.TeacherID
JOIN Disciplines d ON te.SubjectID = d.SubjectID;
```

## 6. Обновление данных

```sql
-- Изменение названия кафедры
UPDATE Teachers 
SET Department = 'Программирования и ВТ' 
WHERE Department = 'ВТиАСУ';

-- Увеличение тарифа на 50% если часов больше 1000
UPDATE Workload 
SET Tariff = Tariff * 1.5 
WHERE HoursPerYear > 1000;

-- Расчет дотации (Subsidy)
UPDATE Workload 
SET Subsidy = 
CASE 
    WHEN Tariff <= 50 THEN Tariff * HoursPerYear * 0.10
    WHEN Tariff > 50 AND Tariff <= 80 THEN Tariff * HoursPerYear * 0.50
    WHEN Tariff > 80 THEN Tariff * HoursPerYear * 0.80
END;

-- Расчет итого (Total)
UPDATE Workload 
SET Total = (Tariff * HoursPerYear) + Subsidy;
```

## 7. Проверка результатов

```sql
-- Просмотр обновленных данных преподавателей
SELECT * FROM Teachers;

-- Просмотр расчетов в нагрузке
SELECT 
    TeacherID,
    Tariff,
    HoursPerYear,
    Subsidy,
    Total,
    (Tariff * HoursPerYear) as BaseSalary
FROM Workload;

-- Итоговый отчет
SELECT 
    t.TeacherID,
    t.FullName,
    t.Department,
    w.Tariff,
    w.HoursPerYear,
    w.Subsidy,
    w.Total
FROM Teachers t
JOIN Workload w ON t.TeacherID = w.TeacherID
ORDER BY t.TeacherID;
```

Все запросы используют английские названия таблиц и полей, соответствуют требованиям задания и включают проверку целостности данных через JOIN-запросы.
