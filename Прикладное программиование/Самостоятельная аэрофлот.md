CREATE DATABASE IF NOT EXISTS aero;
USE aero;

-- Удаление таблиц (в правильном порядке из-за внешних ключей)
DROP TABLE IF EXISTS Pass_in_trip;
DROP TABLE IF EXISTS Trip;
DROP TABLE IF EXISTS Passenger;
DROP TABLE IF EXISTS Company;

-- Создание таблиц
CREATE TABLE Company (
  ID_comp INT NOT NULL,
  name CHAR(10) NOT NULL,
  PRIMARY KEY (ID_comp)
);

CREATE TABLE Passenger (
  ID_psg INT NOT NULL,
  name CHAR(20) NOT NULL,
  PRIMARY KEY (ID_psg)
);

CREATE TABLE Trip (
  trip_no INT NOT NULL,
  ID_comp INT NOT NULL,
  plane CHAR(10) NOT NULL,
  town_from CHAR(25) NOT NULL,
  town_to CHAR(25) NOT NULL,
  time_out DATETIME NOT NULL,
  time_in DATETIME NOT NULL,
  PRIMARY KEY (trip_no),
  FOREIGN KEY (ID_comp) REFERENCES Company(ID_comp)
);

CREATE TABLE Pass_in_trip (
  trip_no INT NOT NULL,
  date DATETIME NOT NULL,
  ID_psg INT NOT NULL,
  place CHAR(10) NOT NULL,
  PRIMARY KEY (trip_no, date, ID_psg),
  FOREIGN KEY (trip_no) REFERENCES Trip(trip_no),
  FOREIGN KEY (ID_psg) REFERENCES Passenger(ID_psg)
);

Для получения нынешнего времени и даты
```mysql
select current_timestamp
```

Для получения года
```mysql
select year(current_timestamp)
```

Для получения дня
```mysql
select day(current_timestamp)
```

Для получения месяца 
```mysql
select month(current_timestamp)
```

DATE_ADD(date, INTERVAL value addunit). **добавляет интервал времени или даты к заданной дате** 
**Параметры**:
- **date** — дата, к которой прибавляется интервал.
- **value** — значение интервала времени или даты для добавления (допускаются как положительные, так и отрицательные значения). 
- **addunit** — тип добавляемого интервала (например, SECOND, MINUTE, HOUR, DAY, YEAR, MONTH и др.)
```mysql
SELECT DATE_ADD('2021-03-01', INTERVAL 10 DAY);
```

`DATEDIFF()` возвращает разницу между двумя датами в днях.
```mysql
select datediff('2026-01-01', current_timestamp);
```

Извлечение дня недели
1. DAYOFWEEK() - возвращает номер дня недели (1=воскресенье, 2=понедельник, ..., 7=суббота)
```sql
SELECT DAYOFWEEK(CURRENT_DATE) AS day_number;
```

2. DAYNAME() - возвращает название дня недели на английском
```sql
SELECT DAYNAME(CURRENT_DATE) AS day_name;
```

3. DATE_FORMAT() - возвращает номер дня недели (0=понедельник, 1=вторник, ..., 6=воскресенье)
```sql
SELECT DATE_FORMAT(CURRENT_DATE, '%w') AS day_number_monday_start;
```

4. DATE_FORMAT() - возвращает название дня недели
```sql
SELECT DATE_FORMAT(CURRENT_DATE, '%W') AS day_name_full;
```

5. WEEKDAY() - возвращает номер дня недели (0=понедельник, 1=вторник, ..., 6=воскресенье)
```sql
SELECT WEEKDAY(CURRENT_DATE) AS weekday_number;
```


Определить количество дней в этом году:
```mysql
select datediff('2025-12-31', '2025-01-01');
```

Определить текущий месяц и время года:
```mysql
SELECT month(CURRENT_TIMESTAMP), CASE WHEN month(CURRENT_TIMESTAMP) in (12, 1, 2) THEN 'Зима' WHEN month(CURRENT_TIMESTAMP) in (3, 4, 5) THEN 'Весна' WHEN month(CURRENT_TIMESTAMP) in (6, 7, 8) THEN 'Лето'else 'Осень' end AS 'Время года'
```
